# Golang vim config

# Install
Step1:
    cd ~
    git clone git@github.com:Lu-Kye/.my_vim.git

Step2:
    git clone https://github.com/gmarik/Vundle.vim.git ~/.vim/bundle/Vundle.vim

Step3:
    echo source ~/.my_vim/my_vimrc.vim >> ~/.vimrc   

Step4:
    open vim and print ":PluginInstall"
# END Install
